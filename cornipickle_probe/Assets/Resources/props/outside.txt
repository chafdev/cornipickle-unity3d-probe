# x-y coordinates of all elements are relative to window
We say that $x extends outside $y when (
 (
  ($x's right is greater than $y's width )
  Or
  ($x's left is less than $y's width)
 )
 Or
 (
  ($x's bottom is greater than $y's height)
  Or
  ($x's top is less than $y's height)
 )
).

"""
  @name Outside window
  @description Coloured squares should not extend outside
    the browser's window
  @severity Error
"""
For each $x in $(Text) (
  For each $y in $(window) (
    $x extends outside $y
  )
).
